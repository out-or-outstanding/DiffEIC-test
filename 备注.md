### inference
```bash

 --ckpt_sd ./weight/v2-1_512-ema-pruned.ckpt \
 --ckpt_lc ./weight/1_2_1/lc.ckpt \
 --config configs/model/diffeic.yaml \
 --input ./images/test/ \

 --steps 50 --device cuda
```
